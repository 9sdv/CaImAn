453 def process_eyelid_traces(traces,time_vect,idx_CS_US,idx_US,idx_CS,thresh_CR=.1,time_CR_on=-.1,time_US_on=.05):
454  """
455  preprocess traces output of get_behavior_traces
456 
457  Parameters:
458  ----------
459 
460  traces: ndarray (N trials X t time points)
461  eyelid traces output of get_behavior_traces.
462 
463  thresh_CR: float
464  fraction of eyelid closure considered a CR
465 
466  time_CR_on: float
467  time of alleged beginning of CRs
468 
469  time_US_on: float
470  time when US is considered to induce have a UR
471 
472 
473  Returns:
474  -------
475  eye_traces: ndarray
476  normalized eyelid traces
477 
478  trigs: dict
479  dictionary containing various subdivision of the triggers according to behavioral responses
480 
481  'idxCSUSCR': index of trials with CS+US with CR
482  'idxCSUSNOCR': index of trials with CS+US without CR
483  'idxCSCR':
484  'idxCSNOCR':
485  'idxNOCR': index of trials with no CRs
486  'idxCR': index of trials with CRs
487  'idxUS':
488 
489  """
490  #normalize by max amplitudes at US
491 
492  eye_traces=old_div(traces,np.nanmax(np.nanmedian(traces[np.hstack([idx_CS_US,idx_US])][:,np.logical_and(time_vect>time_US_on,time_vect<time_US_on +.4 )],0)))
493 
494  amplitudes_at_US=np.mean(eye_traces[:,np.logical_and( time_vect > time_CR_on , time_vect <= time_US_on )],1)
495 
496  trigs=dict()
497 
498  trigs['idxCSUSCR']=idx_CS_US[np.where(amplitudes_at_US[idx_CS_US]>thresh_CR)[-1]]

662 def distance_masks(M_s,cm_s,max_dist):
663  """
664  Compute distance matrix based on an intersection over union metric. Matrix are compared in order, with matrix i compared with matrix i+1
665 
666  Parameters
667  ----------
668  M_s: list of ndarrays
669  The thresholded A matrices (masks) to compare, output of threshold_components
670 
671  cm_s: list of list of 2-ples
672  the centroids of the components in each M_s
673 
674  max_dist: float
675  maximum distance among centroids allowed between components. This corresponds to a distance at which two components are surely disjoined
676 
677 
678 
679  Returns:
680  --------
681  D_s: list of matrix distances
682  """
683  D_s=[]
684 
685  for M1,M2,cm1,cm2 in zip(M_s[:-1],M_s[1:],cm_s[:-1],cm_s[1:]):
686  print('New Pair **')
687  M1= M1.copy()[:,:]
688  M2= M2.copy()[:,:]
689  d_1=np.shape(M1)[-1]
690  d_2=np.shape(M2)[-1]
691  D = np.ones((d_1,d_2));
692 
693  cm1=np.array(cm1)
694  cm2=np.array(cm2)
695  for i in range(d_1):
696  if i%100==0:
697  print(i)
698  k=M1[:,np.repeat(i,d_2)]+M2
699  # h=M1[:,np.repeat(i,d_2)].copy()
700  # h.multiply(M2)
701  for j in range(d_2):
702 
703  dist = np.linalg.norm(cm1[i]-cm2[j])
704  if dist<max_dist:
705  union = k[:,j].sum()
706  # intersection = h[:,j].nnz
707  intersection= np.array(M1[:,i].T.dot(M2[:,j]).todense()).squeeze()
708 
710  if union > 0:
711  D[i,j] = 1-1.*intersection/(union-intersection)
712  else:
713 # print 'empty component: setting distance to max'
714  D[i,j] = 1.
715 
716  if np.isnan(D[i,j]):
717  raise Exception('Nan value produced. Error in inputs')
718  else:
719  D[i,j] = 1
720 
721  D_s.append(D)
722  return D_s
