448 def mode(inputData, axis=None, dtype=None):
449 
450  """
451  Robust estimator of the mode of a data set using the half-sample mode.
452 
453  .. versionadded: 1.0.3
454  """
455  import numpy
456  if axis is not None:
457  fnc = lambda x: mode(x, dtype=dtype)
458  dataMode = numpy.apply_along_axis(fnc, axis, inputData)
459  else:
460  # Create the function that we can use for the half-sample mode
461  def _hsm(data):
462  if data.size == 1:
463  return data[0]
464  elif data.size == 2:
465  return data.mean()
466  elif data.size == 3:
467  i1 = data[1] - data[0]
468  i2 = data[2] - data[1]
469  if i1 < i2:
470  return data[:2].mean()
471  elif i2 > i1:
472  return data[1:].mean()
473  else:
474  return data[1]
475  else:
476 # wMin = data[-1] - data[0]
477  wMin=np.inf
478  N = data.size/2 + data.size%2
479  for i in xrange(0, N):
480  w = data[i+N-1] - data[i]
481  if w < wMin:
482  wMin = w
483  j = i
484 
485  return _hsm(data[j:j+N])
486 
487  data = inputData.ravel()
488  if type(data).__name__ == "MaskedArray":
489  data = data.compressed()
490  if dtype is not None:
491  data = data.astype(dtype)
492 
493  # The data need to be sorted for this to work
494  data = numpy.sort(data)
495 
496  # Find the mode
497  dataMode = _hsm(data)
498 
499  return dataMode

502 def std(inputData, Zero=False, axis=None, dtype=None):
503  """
504  Robust estimator of the standard deviation of a data set.
505 
506  Based on the robust_sigma function from the AstroIDL User's Library.
507 
508  .. versionchanged:: 1.0.3
509  Added the 'axis' and 'dtype' keywords to make this function more
510  compatible with numpy.std()
511  """
512 
513  if axis is not None:
514  fnc = lambda x: std(x, dtype=dtype)
515  sigma = numpy.apply_along_axis(fnc, axis, inputData)
516  else:
517  data = inputData.ravel()
518  if type(data).__name__ == "MaskedArray":
519  data = data.compressed()
520  if dtype is not None:
521  data = data.astype(dtype)
522 
523  if Zero:
524  data0 = 0.0
525  else:
526  data0 = numpy.median(data)
527  maxAbsDev = numpy.median(numpy.abs(data-data0)) / 0.6745
528  if maxAbsDev < __epsilon:
529  maxAbsDev = (numpy.abs(data-data0)).mean() / 0.8000
530  if maxAbsDev < __epsilon:
531  sigma = 0.0
532  return sigma
533 
534  u = (data-data0) / 6.0 / maxAbsDev
535  u2 = u**2.0
536  good = numpy.where( u2 <= 1.0 )
537  good = good[0]
538  if len(good) < 3:
539  print "WARNING: Distribution is too strange to compute standard deviation"
540  sigma = -1.0
541  return sigma
542 
543  numerator = ((data[good]-data0)**2.0 * (1.0-u2[good])**2.0).sum()
544  nElements = (data.ravel()).shape[0]
545  denominator = ((1.0-u2[good])*(1.0-5.0*u2[good])).sum()
546  sigma = nElements*numerator / (denominator*(denominator-1.0))
547  if sigma > 0:
548  sigma = math.sqrt(sigma)
549  else:
550  sigma = 0.0
551 
552  return sigma
